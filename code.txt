#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    float Y, x, myVar;
    cout << "Enter x: ";
    cin >> x;
    myVar = 0;
    for (int k = 1; k <= 5; k++)
    {
        myVar += k;
    }
    myVar += pow(x, 4);
    Y = (tan(x) + 5) / myVar;
    cout << "Y = " << Y << endl;
    return 0;
}
